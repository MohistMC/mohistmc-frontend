    ```java
    @Override
    public void onEnable() {
        // Register an event listener with a custom IEventBus (using Pixelmon's EVENT_BUS as an example here)
        registerForgeEvent(Pixelmon.EVENT_BUS, this);

        // The following principle is the same as bukkit event registration
        // Register an event listener with the default IEventBus (current class)
        registerForgeEvent(this);
        // Registering an event listener with the default IEventBus (external class)
        registerForgeEvent(this, new TestListener());
  }

    @Override
    public void onDisable() {
        // Log out of the specified event listener in the specified IEventBus
        unregisterForgeEvents(IEventBus bus, Object target);
        // Log out of all event listeners in the specified IEventBus
        unregisterForgeEvents(IEventBus bus);
        // Log out of all Forge event listeners for the current plugin
        unregisterAllForgeEvents();
    }
```
