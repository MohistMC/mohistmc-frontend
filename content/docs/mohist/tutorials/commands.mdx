```yaml
#为指定世界创建备份
/backupworld <世界名称>
#增加/展示被ban的 <物品><实体><附魔>
/bans <add|show> <item><entity><enchantment>
#在本地/网页生成 <药水效果|enchants|cbcmds|modscmds|entitytypes|世界群系|pattern|worldgen|世界类型|物品NBT|channels|成就>
/dump <file|web> <potions|enchants|cbcmds|modscmds|entitytypes|biomes|pattern|worldgen|worldtype|material|channels|advancements>
#获取服务器插件列表
/getpluginlist
#(手持)物品[获取信息|设置名称|另存为|列表|获取|移除]
/items [info|name|save|list|get|remove]
#查看服务器 [模组玩家客户端模组|重载|版本|channels_incom|channels_outgo|设置玩家速度(最大为10)]
/mohist [mods|playermods|reload|version|channels_incom|channels_outgo|speed]
#查看玩家延迟
/ping [player_name]
#查看玩家权限
/permission <check>
#展示[声音|实体|物品]
/shows [sound|entitys|blockentitys]
#tpa玩家
/tpa <player_name
#同意玩家的tpa
/tpaccept
#拒绝玩家的tpa
/tpadeny
#设置一个地标
/warps set <Name>
#删除一个地标
/warps del <Name>
#传送到某个地标
/warps tp  <Name>
#将玩家传送至某个地标
/warps tp <Player> <Name>
#打开地标列表
/warps gui
#创建一个<Name>世界
/worlds create <Name>
#删除一个<Name>世界
/worlds delete <Name>
#传送到<Name>世界
/worlds tp <Name>
#将玩家传送至<Name>世界
/worlds tp <Player> <Name>
#打开世界列表
/worlds gui
#设置世界出生点
/worlds setspawn
#导入一个<Name>世界
/worlds import <Name>
#卸载一个<Name>世界
/worlds unload <Name>
#<Name>世界的信息
/worlds info <Name>
#给<Name>世界加入额外信息
/worlds addinfo <Name>
#为<Name>世界设置名称
/worlds setname <Name>
#设置世界的难度
/worlds difficulty <0-3>
```
