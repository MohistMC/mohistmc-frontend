```yaml
#Create backup for specified world
/backupworld <world_name>
#Add/Show banned <items><entities><enchantments>
/bans <add|show> <item><entity><enchantment>
#Generate <potions|enchants|cbcmds|modscmds|entitytypes|world_biomes|pattern|worldgen|world_type|item_NBT|channels|achievements> locally/web
/dump <file|web> <potions|enchants|cbcmds|modscmds|entitytypes|biomes|pattern|worldgen|worldtype|material|channels|advancements>
#Get server plugin list
/getpluginlist
#(Held) item [get info|set name|save as|list|get|remove]
/items [info|name|save|list|get|remove]
#View server [mods|player client mods|reload|version|channels_incom|channels_outgo|set player speed (max 10)]
/mohist [mods|playermods|reload|version|channels_incom|channels_outgo|speed]
#Check player ping
/ping [player_name]
#Check player permissions
/permission <check>
#Show [sounds|entities|items]
/shows [sound|entitys|blockentitys]
#TPA player
/tpa <player_name>
#Accept player's TPA
/tpaccept
#Deny player's TPA
/tpadeny
#Set a landmark
/warps set <Name>
#Delete a landmark
/warps del <Name>
#Teleport to a landmark
/warps tp <Name>
#Teleport player to a landmark
/warps tp <Player> <Name>
#Open landmark list
/warps gui
#Create a <Name> world
/worlds create <Name>
#Delete a <Name> world
/worlds delete <Name>
#Teleport to <Name> world
/worlds tp <Name>
#Teleport player to <Name> world
/worlds tp <Player> <Name>
#Open world list
/worlds gui
#Set world spawn point
/worlds setspawn
#Import a <Name> world
/worlds import <Name>
#Unload a <Name> world
/worlds unload <Name>
#<Name> world information
/worlds info <Name>
#Add extra information to <Name> world
/worlds addinfo <Name>
#Set name for <Name> world
/worlds setname <Name>
#Set world difficulty
/worlds difficulty <0-3>
```
